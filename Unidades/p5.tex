\subsection{Definiciones básicas y primeras propiedades}

\begin{defi}[Conjunto compacto]
Sea $(E,d)$ un espacio métrico y $K \subseteq E$. Decimos que $K$ es
\emph{compacto} si para toda sucesión $(x_n)_{n\in\N} \subseteq K$ existe
una subsucesión $(x_{n_k})_{k\in\N}$ y un punto $x \in K$ tales que
\[
x_{n_k} \xrightarrow[k\to\infty]{} x.
\]
\end{defi}

\begin{defi}[Punto de acumulación]
Sea $(E,d)$ un espacio métrico y $A \subseteq E$.
Un punto $x \in E$ se llama \emph{punto de acumulación} (o punto límite)
de $A$ si
\[
\forall r>0 \quad \bigl(B(x,r)\setminus\{x\}\bigr) \cap A \neq \varnothing.
\]
Denotamos por $A'$ al conjunto de puntos de acumulación de $A$.
\end{defi}

\begin{prop}[Caracterización secuencial de puntos de acumulación]
Sea $A \subseteq E$ y $x \in E$. Entonces
\[
x \in A' \iff \exists (a_n)_{n\in\N} \subseteq A\setminus\{x\}
\text{ tal que } a_n \to x.
\]
\end{prop}

\begin{prop}[Compacidad implica cerrado y acotado]
Sea $(E,d)$ un espacio métrico y $K \subseteq E$ compacto. Entonces $K$ es
cerrado y acotado.
\end{prop}

\begin{proof}
\textbf{Cerrado.} Sea $x \in \overline{K}$. Existe $(x_n)_{n\in\N}\subseteq K$
tal que $x_n \to x$. Como $K$ es compacto, existe una subsucesión
$(x_{n_k})$ y un punto $y\in K$ tales que $x_{n_k}\to y$. Pero como la
sucesión completa $(x_n)$ converge a $x$, toda subsucesión suya también
converge a $x$; por unicidad del límite, $x=y \in K$. Se obtiene
$\overline{K}\subseteq K$, y como siempre $K\subseteq\overline{K}$,
concluimos $\overline{K}=K$, es decir, $K$ es cerrado.

\textbf{Acotado.} Recordemos que $K$ es acotado si existe $x\in K$ y $C>0$
tales que $K\subseteq B(x,C)$. Supongamos, por el absurdo, que $K$ no es
acotado, y elijamos $x_1\in K$ cualquiera. Como $K\not\subseteq B(x_1,1)$,
existe $x_2\in K$ tal que $d(x_1,x_2)\ge 1$. Como $K\not\subseteq
B(x_1,2)$, existe $x_3\in K$ tal que $d(x_1,x_3)\ge 2$. Recursivamente,
podemos elegir $x_n\in K$ tal que $d(x_1,x_n)\ge n-1$ para todo $n\in\N$.

Como $K$ es compacto, existe una subsucesión $(x_{n_k})_{k\in\N}$ y
$x\in K$ tales que $x_{n_k}\to x$. Consideremos la función
$f:E\to\R$, $f(y)=d(x_1,y)$. Por la desigualdad triangular al revés,
\[
|f(y)-f(z)| = |d(x_1,y)-d(x_1,z)| \le d(y,z) \quad \forall y,z\in E,
\]
por lo que $f$ es (1-)Lipschitz y en particular continua. Entonces
$f(x_{n_k})\to f(x) = d(x_1,x) \in \R$.

Pero por construcción,
\[
f(x_{n_k}) = d(x_1,x_{n_k}) \ge n_k - 1 \xrightarrow[k\to\infty]{} \infty,
\]
lo cual contradice que $f(x_{n_k})$ converja a un número real. Este
absurdo muestra que $K$ debe ser acotado.
\end{proof}

\begin{lema}[Bolzano--Weierstrass en $\R$]
Toda sucesión acotada de números reales admite una subsucesión
convergente.
\end{lema}

\begin{proof}
Sea $(a_n)_{n\in\N}$ una sucesión acotada de reales. Veamos primero que
$(a_n)$ admite una subsucesión monótona. Decimos que un índice $n$ es un
\emph{pico} de la sucesión si $a_n \ge a_m$ para todo $m>n$.

Si hay infinitos picos $n_1<n_2<n_3<\dots$, entonces por definición de
pico se cumple $a_{n_1}\ge a_{n_2}\ge a_{n_3}\ge \dots$, es decir,
$(a_{n_k})_{k\in\N}$ es una subsucesión monótona decreciente.

Si hay sólo finitos picos, sea $N$ mayor que todos ellos. Tomamos
$n_1 = N+1$; como $n_1$ no es un pico, existe $n_2>n_1$ con
$a_{n_2}>a_{n_1}$. Como $n_2>N$ tampoco es un pico, existe $n_3>n_2$ con
$a_{n_3}>a_{n_2}$. Repitiendo este proceso obtenemos una subsucesión
estrictamente creciente $a_{n_1}<a_{n_2}<a_{n_3}<\dots$

En cualquier caso, obtuvimos una subsucesión monótona $(a_{n_k})_{k\in\N}$
de $(a_n)$. Como $(a_n)$ está acotada, también lo está $(a_{n_k})$. Por
la proposición de convergencia de sucesiones monótonas y acotadas
(aplicada según sea creciente o decreciente), $(a_{n_k})$ converge.
\end{proof}

\begin{obs}
En un espacio métrico general, un conjunto cerrado y acotado no es
necesariamente compacto (más adelante, en espacios de sucesiones, se
pueden construir contraejemplos). Sin embargo, cuando el espacio ambiente
es $(\R^m,d_2)$, sí vale la recíproca de la proposición anterior.
\end{obs}

\begin{teo}[Heine--Borel en $\R^m$]
Sea $K \subseteq \R^m$ con la métrica euclídea usual, cerrado y acotado.
Entonces $K$ es compacto.
\end{teo}

\begin{proof}
Empecemos por el caso $m=1$, es decir, $K\subseteq \R$.

Sea $(x_n)_{n\in\N} \subseteq K$ una sucesión. Como $K$ es acotado,
$(x_n)$ es una sucesión acotada de reales, y por el lema de
Bolzano--Weierstrass admite una subsucesión convergente $x_{n_k}\to x \in
\R$. Como $K$ es cerrado y $(x_{n_k})\subseteq K$, se tiene $x\in K$. Por
lo tanto $K$ es compacto.

Sea ahora $m>1$ y $(x_n)_{n\in\N}\subseteq K \subseteq \R^m$, con
$x_n = (x_n^1,\dots,x_n^m)$. Como $K$ es acotado, existe $C>0$ tal que
$d_2(x_n,x_1)\le C$ para todo $n$. Para cada coordenada $i$,
\[
|x_n^i - x_1^i| = \sqrt{(x_n^i-x_1^i)^2} \le
\sqrt{\sum_{j=1}^m (x_n^j-x_1^j)^2} = d_2(x_n,x_1) \le C,
\]
de modo que $(x_n^i)_{n\in\N}$ es una sucesión acotada de reales para
cada $i=1,\dots,m$.

Aplicamos el lema de Bolzano--Weierstrass a la primera coordenada: existe
una subsucesión $(x_{n_{k_1}}^1)_{k_1\in\N}$ que converge a algún $x^1 \in
\R$. Consideramos ahora la sucesión de segundas coordenadas restringida a
esos índices, $(x_{n_{k_1}}^2)_{k_1\in\N}$, que sigue siendo acotada; por
el lema, admite una subsucesión $(x_{n_{k_1 k_2}}^2)_{k_2}$ convergente a
algún $x^2\in\R$. Repitiendo este proceso para cada una de las $m$
coordenadas (tomando en cada paso subsucesión de la subsucesión
anterior), obtenemos al final una subsucesión $(x_{n_k})_{k\in\N}$ de
$(x_n)$ tal que
\[
x_{n_k}^i \to x^i \quad \text{para todo } i=1,\dots,m.
\]

Sea $x=(x^1,\dots,x^m)$. Dado $\varepsilon>0$, para cada $i$ existe $k_i$
tal que $|x_{n_k}^i - x^i| < \varepsilon/\sqrt{m}$ para todo $k\ge k_i$.
Tomando $k_0 = \max\{k_1,\dots,k_m\}$, para $k\ge k_0$,
\[
d_2(x_{n_k},x) = \sqrt{\sum_{i=1}^m (x_{n_k}^i-x^i)^2}
< \sqrt{m \cdot \frac{\varepsilon^2}{m}} = \varepsilon,
\]
es decir, $x_{n_k}\to x$ en $(\R^m,d_2)$.

Como $K$ es cerrado y $(x_{n_k})\subseteq K$, se tiene $x\in K$. Por lo
tanto $(x_n)$ admite una subsucesión convergente en $K$, es decir, $K$ es
compacto.
\end{proof}

\begin{obs}
Combinando la proposición de compacidad implica cerrado y acotado con el
Teorema de Heine--Borel, en $(\R^m,d_2)$ vale la equivalencia
\[
K \text{ es compacto} \iff K \text{ es cerrado y acotado}.
\]
\end{obs}

\subsection{Propiedades estructurales de los compactos}

\begin{prop}[Subconjunto cerrado de un compacto]
Sea $(E,d)$ un espacio métrico, $K \subseteq E$ compacto y $F \subseteq K$
cerrado. Entonces $F$ es compacto.
\end{prop}

\begin{proof}
Sea $(x_n)_{n\in\N} \subseteq F$ una sucesión. Como $F\subseteq K$ y $K$
es compacto, existe una subsucesión $(x_{n_k})_{k\in\N}$ y $x\in K$ tales
que $x_{n_k}\to x$. Como $F$ es cerrado y $(x_{n_k})\subseteq F$, se
tiene $x\in F$. Por lo tanto $F$ es compacto.
\end{proof}

\begin{prop}[Uniones e intersecciones de compactos]
Sea $(E,d)$ un espacio métrico y $\{K_i\}_{i\in I}$ una familia de
subconjuntos compactos. Entonces:
\begin{enumerate}[label=(\roman*)]
    \item $\displaystyle\bigcup_{i=1}^m K_i$ es compacto, para cualquier
    subfamilia finita $K_1,\dots,K_m$.
    \item $\displaystyle\bigcap_{i\in I} K_i$ es compacto.
\end{enumerate}
\end{prop}

\begin{proof}
(i) Sea $(x_n)_{n\in\N} \subseteq \bigcup_{i=1}^m K_i$ una sucesión. Como
la unión es finita, alguno de los $K_i$, digamos $K_{i_0}$, contiene
infinitos términos de la sucesión, es decir, existe una subsucesión
$(x_{n_k})_{k\in\N} \subseteq K_{i_0}$. Como $K_{i_0}$ es compacto, existe
una subsucesión $(x_{n_{k_j}})_{j\in\N}$ y $x\in K_{i_0}$ tales que
$x_{n_{k_j}}\to x$. Como $x\in K_{i_0}\subseteq \bigcup_{i=1}^m K_i$,
concluimos que $\bigcup_{i=1}^m K_i$ es compacto.

(ii) Por la proposición anterior, cada $K_i$ es cerrado (todo compacto es
cerrado), y la intersección arbitraria de cerrados es cerrada, así que
$\bigcap_{i\in I} K_i$ es cerrado. Además, fijado cualquier $i_0\in I$,
\[
\bigcap_{i\in I} K_i \subseteq K_{i_0},
\]
y $K_{i_0}$ es compacto. Por la proposición de subconjunto cerrado de un
compacto, $\bigcap_{i\in I} K_i$ es compacto.
\end{proof}

\begin{obs}
La unión de \emph{infinitos} compactos puede no ser compacta. Por
ejemplo, en $(\R,|\cdot|)$, sean $K_i=[1/i,1]$ para $i\in\N$; cada $K_i$
es compacto (cerrado y acotado), pero
\[
\bigcup_{i=1}^\infty K_i = (0,1],
\]
que no es cerrado y por lo tanto no es compacto.
\end{obs}

\begin{prop}[Compacidad implica completitud]
Sea $(E,d)$ un espacio métrico compacto. Entonces $E$ es completo.
\end{prop}

\begin{proof}
Sea $(x_n)_{n\in\N}$ una sucesión de Cauchy en $E$. Como $E$ es compacto,
existe una subsucesión $(x_{n_k})$ convergente a algún $x\in E$. Pero una
sucesión de Cauchy que admite una subsucesión convergente converge (al
mismo límite que la subsucesión). Por lo tanto $(x_n)$ converge, y $E$ es
completo.
\end{proof}

\subsection{Cubrimientos por abiertos}

\begin{defi}[Cubrimiento por abiertos]
Sea $(E,d)$ un espacio métrico y $K \subseteq E$. Una familia
$\{U_i\}_{i\in I}$ de subconjuntos abiertos de $E$ es un
\emph{cubrimiento por abiertos} de $K$ si
\[
K \subseteq \bigcup_{i \in I} U_i.
\]
Dado un cubrimiento por abiertos $\{U_i\}_{i\in I}$ de $K$, un
\emph{subcubrimiento} es una subfamilia $\{U_i\}_{i\in J}$, con
$J\subseteq I$, tal que $K\subseteq \bigcup_{i\in J} U_i$. Si $J$ es
finito, decimos que es un \emph{subcubrimiento finito}.
\end{defi}

\begin{teo}[Caracterizaciones de la compacidad]
Sea $(E,d)$ un espacio métrico y $K \subseteq E$. Son equivalentes:
\begin{enumerate}[label=(\roman*)]
    \item $K$ es compacto.
    \item Todo subconjunto infinito $A \subseteq K$ tiene al menos un
    punto de acumulación en $K$, es decir, $A' \cap K \neq \varnothing$.
    \item Todo cubrimiento por abiertos de $K$ admite un subcubrimiento
    finito.
\end{enumerate}
\end{teo}

\begin{proof}
$(i) \Rightarrow (ii)$.
Sea $A\subseteq K$ infinito. Elegimos una sucesión de puntos distintos
$(x_n)\subseteq A$ (por ejemplo, una enumeración inyectiva de $A$).
Como $K$ es compacto, existe una subsucesión $(x_{n_j})$ y un punto
$x\in K$ tales que $x_{n_j}\to x$. Como todos los $x_{n_j}\in A$, la
caracterización secuencial de puntos de acumulación dice que $x\in A'$.
Por tanto $A'\cap K\neq \varnothing$.

\medskip

$(ii) \Rightarrow (i)$.
Sea $(x_n)$ una sucesión en $K$ y sea $A=\{x_n:n\in\N\}\subseteq K$.
Si $A$ es finito, la sucesión $(x_n)$ debe tomar algún valor infinitas
veces, y esos términos forman una subsucesión constante, luego
convergente.

Si $A$ es infinito, por (ii) existe $x\in A'\cap K$. Entonces, por la
caracterización secuencial de puntos de acumulación, existe una
subsucesión $(x_{n_j})$ de $(x_n)$ tal que $x_{n_j}\to x\in K$.
En todo caso, $(x_n)$ admite una subsucesión convergente en $K$, es
decir, $K$ es compacto.

\medskip

$(iii) \Rightarrow (ii)$.
Sea $A\subseteq K$ un subconjunto infinito y supongamos, por el absurdo,
que $A$ no tiene puntos de acumulación en $K$. Esto significa que para
cada $x\in K$, $x$ no es punto de acumulación de $A$, por lo que existe
un radio $r_x>0$ tal que $B(x,r_x)\cap A$ es finito. La familia
$\{B(x,r_x)\}_{x\in K}$ es un cubrimiento por abiertos de $K$, y por
(iii) admite un subcubrimiento finito:
\[
K \subseteq \bigcup_{k=1}^n B(x_k,r_{x_k}).
\]
Como $A\subseteq K$,
\[
A = A\cap K = \bigcup_{k=1}^n \bigl(A\cap B(x_k,r_{x_k})\bigr),
\]
y por lo tanto
\[
\#A \le \sum_{k=1}^n \#\bigl(A\cap B(x_k,r_{x_k})\bigr) < \infty,
\]
al ser una suma finita de conjuntos finitos. Esto contradice que $A$ sea
infinito. Por lo tanto $A$ tiene un punto de acumulación en $K$.

\medskip

$(i) \Rightarrow (iii)$.
Probamos la contrarrecíproca. Supongamos que $K$ no satisface (iii).
Entonces existe un recubrimiento abierto $\{G_i\}_{i\in I}$ de $K$
que no admite subcubrimiento finito.

Construimos inductivamente una sucesión $(x_n)$ en $K$ del siguiente modo:

- Elegimos $x_1\in K$ y un índice $i_1$ con $x_1\in G_{i_1}$.
- Supuesto elegido $x_1,\dots,x_n$ con índices $i_1,\dots,i_n$ tales
  que $x_k\in G_{i_k}$, como
  \(
    K \nsubseteq \bigcup_{k=1}^n G_{i_k}
  \)
  existe
  \(
    x_{n+1}\in K \setminus \bigcup_{k=1}^n G_{i_k}.
  \)
  Elegimos $i_{n+1}$ con $x_{n+1}\in G_{i_{n+1}}$.

Sea $(x_{n_j})$ una subsucesión cualquiera de $(x_n)$.
Si fuera convergente, digamos $x_{n_j}\to x\in K$, algún abierto $G_i$
contendría a $x$ y a todos los términos suficientemente avanzados de
la subsucesión; en particular, ese solo $G_i$ cubriría casi todos los
$x_n$, y un número finito de abiertos de la familia cubriría $K$,
contradiciendo la elección de $\{G_i\}$ sin subcubrimiento finito.
Luego $(x_n)$ no posee subsucesión convergente, y $K$ no es compacto.

La contrarrecíproca muestra que (i) implica (iii). Junto con las
implicaciones anteriores, quedan probadas las tres equivalencias.
\end{proof}

\subsection{Funciones continuas sobre compactos}

\begin{teo}[Imagen continua de un compacto]
Sean $(E,d)$ y $(E',d')$ espacios métricos y sea
$f : E \to E'$ continua. Si $K \subseteq E$ es compacto, entonces
$f(K)$ es compacto en $E'$.
\end{teo}

\begin{proof}
Sea $\{V_\alpha\}_{\alpha\in A}$ un recubrimiento abierto de $f(K)$ en $E'$.
Para cada $\alpha$ definimos $U_\alpha = f^{-1}(V_\alpha)$, que es abierto
en $E$ por continuidad de $f$. Además,
\[
K \subseteq \bigcup_{\alpha\in A} U_\alpha,
\]
pues si $x\in K$ entonces $f(x)\in f(K)\subseteq \bigcup V_\alpha$.

Como $K$ es compacto, existe un subcubrimiento finito
$K \subseteq \bigcup_{j=1}^n U_{\alpha_j}$. Aplicando $f$,
\[
f(K) \subseteq \bigcup_{j=1}^n f(U_{\alpha_j})
\subseteq \bigcup_{j=1}^n V_{\alpha_j},
\]
que es un subcubrimiento finito de $f(K)$. Así, $f(K)$ es compacto.
\end{proof}

\begin{obs}
La demostración anterior también puede hacerse directamente a partir de
la definición de compacidad por sucesiones, sin pasar por cubrimientos:
si $(y_n)_{n\in\N}\subseteq f(K)$, escribimos $y_n=f(x_n)$ con
$x_n\in K$; como $K$ es compacto existe una subsucesión $x_{n_k}\to x\in
K$, y por continuidad $f(x_{n_k})\to f(x)\in f(K)$, lo que da la
subsucesión convergente buscada.
\end{obs}

\begin{cor}[Teorema de Weierstrass]
Sea $K \subseteq E$ compacto y $f : K \to \R$ continua.
Entonces:
\begin{enumerate}[label=(\roman*)]
    \item $f$ es acotada en $K$: existe $c>0$ tal que
    $|f(x)|\le c$ para todo $x\in K$;
    \item $f$ alcanza su máximo y su mínimo en $K$, es decir,
    existen $x_{\min},x_{\max}\in K$ con
    \[
    f(x_{\min}) \le f(x) \le f(x_{\max}) \quad \forall x\in K.
    \]
\end{enumerate}
\end{cor}

\begin{proof}
Por el teorema anterior, $f(K)$ es compacto en $\R$. Todo compacto
no vacío de $\R$ es cerrado y acotado, de modo que admite supremo e
ínfimo que pertenecen al conjunto; eso da el máximo y el mínimo.
La acotación se deduce de la existencia de supremo y de ínfimo.
\end{proof}

\begin{teo}[Continuidad uniforme en compactos]
Sean $(E,d)$ y $(E',d')$ espacios métricos, con $E$ compacto, y sea
$f : E \to E'$ continua. Entonces $f$ es uniformemente continua.
\end{teo}

\begin{proof}
Por la caracterización secuencial de la no continuidad uniforme (Unidad
4), alcanza con ver que para cualesquiera sucesiones $(x_n),(y_n)
\subseteq E$ tales que $d(x_n,y_n)\to 0$, se tiene
$d'(f(x_n),f(y_n))\to 0$.

Supongamos, por el absurdo, que existen sucesiones $(x_n),(y_n)$ con
$d(x_n,y_n)\to 0$ pero $d'(f(x_n),f(y_n)) \not\to 0$. Entonces existe
$\varepsilon>0$ y una subsucesión tal que
\[
d'(f(x_{n_k}),f(y_{n_k})) \ge \varepsilon \quad \text{para todo } k\in\N.
\]
Como $E$ es compacto, existe una subsucesión $(x_{n_{k_j}})_{j\in\N}$ y
$x\in E$ tales que $x_{n_{k_j}}\to x$. Entonces
\[
d(y_{n_{k_j}},x) \le d(y_{n_{k_j}},x_{n_{k_j}}) + d(x_{n_{k_j}},x)
\xrightarrow[j\to\infty]{} 0,
\]
por lo que $y_{n_{k_j}}\to x$ también. Como $f$ es continua,
\[
f(x_{n_{k_j}}) \to f(x) \quad\text{y}\quad f(y_{n_{k_j}}) \to f(x).
\]
Pero entonces
\[
\varepsilon \le d'(f(x_{n_{k_j}}),f(y_{n_{k_j}}))
\le d'(f(x_{n_{k_j}}),f(x)) + d'(f(x),f(y_{n_{k_j}}))
\xrightarrow[j\to\infty]{} 0,
\]
lo cual es absurdo. Por lo tanto $f$ es uniformemente continua.
\end{proof}

\begin{cor}[Funciones continuas sobre compacto son cerradas]
Sean $(E,d)$ y $(E',d')$ espacios métricos, $E$ compacto y
$f : E \to E'$ continua. Entonces $f$ es una aplicación cerrada:
si $G \subseteq E$ es cerrado, entonces $f(G)$ es cerrado en $E'$.
\end{cor}

\begin{proof}
Si $G$ es cerrado en $E$, entonces $G$ es compacto (cerrado en un
compacto). Por el teorema de la imagen de un compacto, $f(G)$ es
compacto en $E'$. En un espacio métrico, todo compacto es cerrado,
así que $f(G)$ es cerrado.
\end{proof}

\begin{cor}[Homeomorfismos desde compactos]
Sean $(E,d)$ y $(E',d')$ espacios métricos, con $E$ compacto, y sea
$f : E \to E'$ continua y biyectiva. Entonces $f$ es un homeomorfismo,
es decir, $f^{-1} : E' \to E$ es continua.
\end{cor}

\begin{proof}
Del corolario anterior, $f$ es una aplicación cerrada.
Sea $F' \subseteq E'$ cerrado. Como $f$ es biyectiva, se tiene
\[
f^{-1}(F') \subseteq E.
\]
Además
\[
E' \setminus F' \text{ es abierto} \Rightarrow
f^{-1}(E'\setminus F') = E \setminus f^{-1}(F')
\text{ es abierto en }E,
\]
por continuidad de $f$. Por lo tanto $f^{-1}(F')$ es cerrado en $E$.
La preimagen por $f^{-1}$ de todo cerrado de $E$ es cerrada, lo que
equivale a la continuidad de $f^{-1}$. Luego $f$ es un homeomorfismo.
\end{proof}

\subsection{Teorema del punto fijo}

\begin{defi}[Punto fijo]
Sea $(E,d)$ un espacio métrico y $f : E \to E$ una función. Un punto
$x_0 \in E$ se dice \emph{punto fijo} de $f$ si $f(x_0)=x_0$.
\end{defi}

\begin{defi}[Contracción]
Sea $(E,d)$ un espacio métrico y $f : E \to E$ una función. Decimos que
$f$ es una \emph{contracción} si existe $0<\alpha<1$ tal que
\[
d(f(x),f(y)) \le \alpha\, d(x,y) \quad \text{para todo } x,y \in E.
\]
\end{defi}

\begin{obs}
Toda contracción es Lipschitz (de constante $\alpha$) y, por lo tanto,
uniformemente continua.
\end{obs}

\begin{teo}[Punto fijo de Banach]
Sea $(E,d)$ un espacio métrico completo y sea $f : E \to E$ una
contracción, de constante $\alpha \in (0,1)$. Entonces $f$ tiene un único
punto fijo $x^* \in E$. Más aún, para cualquier $x_0 \in E$, la sucesión
definida por $x_n = f(x_{n-1})$ converge a $x^*$, con la estimación de
error
\[
d(x_n, x^*) \le \frac{\alpha^n}{1-\alpha}\, d(x_1,x_0)
\quad \text{para todo } n \in \N.
\]
\end{teo}

\begin{proof}
\textbf{Construcción de la sucesión.} Sea $x_0\in E$ cualquiera y
definamos $x_n = f(x_{n-1})$ para $n\ge 1$.

\textbf{Cota para pasos consecutivos.} Veamos que
\[
d(x_n,x_{n+1}) \le \alpha^n\, d(x_0,x_1) \quad \text{para todo } n\in\N.
\]
Por inducción: para $n=0$ es trivial. Si vale para $n-1$, como $f$ es
contracción,
\[
d(x_n,x_{n+1}) = d(f(x_{n-1}),f(x_n)) \le \alpha\, d(x_{n-1},x_n)
\le \alpha \cdot \alpha^{n-1} d(x_0,x_1) = \alpha^n d(x_0,x_1).
\]

\textbf{$(x_n)$ es de Cauchy.} Sean $n<m$ naturales, y escribamos
$m=n+k$. Por la desigualdad triangular y la cota anterior,
\[
d(x_n,x_{n+k}) \le \sum_{j=0}^{k-1} d(x_{n+j},x_{n+j+1})
\le \sum_{j=0}^{k-1} \alpha^{n+j}\, d(x_0,x_1)
= \alpha^n d(x_0,x_1) \sum_{j=0}^{k-1} \alpha^j
\le \alpha^n d(x_0,x_1) \sum_{j=0}^{\infty} \alpha^j
= \frac{\alpha^n}{1-\alpha}\, d(x_0,x_1),
\]
donde usamos que $\sum_{j\ge0}\alpha^j = \frac{1}{1-\alpha}$ por ser
$0<\alpha<1$. Como $\alpha^n \to 0$ cuando $n\to\infty$, dado
$\varepsilon>0$ podemos elegir $n_0$ tal que
$\frac{\alpha^{n_0}}{1-\alpha} d(x_0,x_1) < \varepsilon$; entonces, para
todo $m>n\ge n_0$,
\[
d(x_n,x_m) \le \frac{\alpha^n}{1-\alpha} d(x_0,x_1)
\le \frac{\alpha^{n_0}}{1-\alpha} d(x_0,x_1) < \varepsilon.
\]
Por lo tanto $(x_n)$ es de Cauchy.

\textbf{Convergencia.} Como $(E,d)$ es completo, $(x_n)$ converge a algún
$x^*\in E$.

\textbf{$x^*$ es punto fijo.} Como $f$ es Lipschitz, es continua, así
que $f(x_n)\to f(x^*)$. Pero $f(x_n)=x_{n+1}$, y $(x_{n+1})_{n\in\N}$
converge al mismo límite que $(x_n)_{n\in\N}$, por lo que también
$x_{n+1}\to x^*$. Por unicidad del límite, $f(x^*)=x^*$.

\textbf{Unicidad.} Supongamos que $x^*$ e $y^*$ son ambos puntos fijos de
$f$. Entonces
\[
d(x^*,y^*) = d(f(x^*),f(y^*)) \le \alpha\, d(x^*,y^*).
\]
Como $0<\alpha<1$, esto obliga a $d(x^*,y^*)=0$ (si fuera $d(x^*,y^*)>0$,
dividiendo por $d(x^*,y^*)$ obtendríamos $1\le\alpha<1$, absurdo). Por lo
tanto $x^*=y^*$.

\textbf{Estimación de error.} Tomando $m\to\infty$ en la desigualdad
$d(x_n,x_{n+k})\le \frac{\alpha^n}{1-\alpha}d(x_0,x_1)$ (válida para todo
$k$) obtenemos
\[
d(x_n,x^*) \le \frac{\alpha^n}{1-\alpha}\, d(x_0,x_1),
\]
como se quería.
\end{proof}

\begin{teo}[Punto fijo con compacidad]
Sea $(E,d)$ un espacio métrico compacto y sea $f : E \to E$ una función
continua tal que
\[
d(f(x),f(y)) < d(x,y) \quad \text{para todo } x\neq y \in E.
\]
Entonces $f$ tiene un único punto fijo.
\end{teo}

\begin{proof}
\textbf{Existencia.} Consideremos el conjunto
\[
A = \{d(x,f(x)) : x\in E\} \subseteq \R.
\]
Como $A$ está acotado inferiormente por $0$, existe $i = \inf A$. Veamos
que $i$ se alcanza y que $i=0$.

Sea $(x_n)_{n\in\N}\subseteq E$ tal que $d(x_n,f(x_n)) \to i$ (existe por
la propiedad caracterizadora del ínfimo). Como $E$ es compacto, existe
una subsucesión $(x_{n_k})$ y $x^*\in E$ tales que $x_{n_k}\to x^*$. Como
$f$ es continua, $f(x_{n_k})\to f(x^*)$.

Por la desigualdad triangular,
\[
\bigl|d(x_{n_k},f(x_{n_k})) - d(x^*,f(x^*))\bigr|
\le d(x_{n_k},x^*) + d(f(x_{n_k}),f(x^*)) \xrightarrow[k\to\infty]{} 0,
\]
así que $d(x_{n_k},f(x_{n_k})) \to d(x^*,f(x^*))$. Pero por construcción
$d(x_{n_k},f(x_{n_k})) \to i$, así que por unicidad del límite,
\[
d(x^*,f(x^*)) = i,
\]
es decir, el ínfimo $i$ se alcanza en $x^*$.

Veamos que $i=0$. Supongamos que no, es decir que $y:=f(x^*)\neq x^*$.
Entonces
\[
d(y,f(y)) = d(f(x^*),f(y)) < d(x^*,y) = d(x^*,f(x^*)) = i,
\]
usando la hipótesis sobre $f$ (con $x^*\neq y$). Pero esto da un elemento
de $A$ estrictamente menor que $i=\inf A$, absurdo. Por lo tanto $i=0$,
es decir, $f(x^*)=x^*$.

\textbf{Unicidad.} Si $x^*$ e $y^*$ fueran dos puntos fijos distintos de
$f$, entonces por hipótesis
\[
d(x^*,y^*) = d(f(x^*),f(y^*)) < d(x^*,y^*),
\]
lo cual es absurdo. Por lo tanto el punto fijo es único.
\end{proof}
