\subsection{Axioma de completitud}
Dado $A \subset\mathbb{R}$ no vacío y acotado superiormente, existe $\sup A$.\\
Dado $A \subset\mathbb{R}$ no vacío y acotado inferiormente, existe $\inf A$.

\subsection{Completitud para ínfimos}
La existencia del ínfimo no necesita postularse como un axioma aparte: se deduce
de la existencia del supremo.
\[
\text{Sea } A \subset \R \text{ no vacío y acotado inferiormente. Entonces existe } i = \inf A \in \R.
\]
\begin{proof}
Consideremos el conjunto
\[
-A = \{-a : a \in A\}.
\]
Como $A \neq \emptyset$, también $-A \neq \emptyset$. Veamos que $-A$ está acotado
superiormente: sea $c$ una cota inferior de $A$, es decir $c \le a$ para todo
$a \in A$. Multiplicando por $-1$ obtenemos $-c \ge -a$ para todo $a \in A$, es
decir, $-c$ es cota superior de $-A$. Por el axioma de completitud (para
supremos), existe $s = \sup(-A)$.

Afirmamos que $i = -s$ es el ínfimo de $A$.

(i) $i$ es cota inferior de $A$: como $s$ es cota superior de $-A$, se cumple
$-a \le s$ para todo $a \in A$, es decir, $-s \le a$ para todo $a \in A$, o sea
$i \le a$ para todo $a \in A$.

(ii) $i$ es la mayor de las cotas inferiores de $A$: sea $c$ una cota inferior
de $A$. Como vimos, $-c$ es cota superior de $-A$, y como $s = \sup(-A)$ es la
menor de las cotas superiores de $-A$, se cumple $s \le -c$, es decir,
$c \le -s = i$.

De (i) y (ii) concluimos que $i = -s = \inf A$.
\end{proof}

\subsection{Infimo}
Sea $A \subset\mathbb{R}$, no vacío y acotado inferiormente.
\[
i = \inf A \Longleftrightarrow
\begin{cases}
i \le a & \text{para todo } a \in A,\\[4pt]
\forall \varepsilon > 0 \ \exists a \in A \text{ tal que } i \le a < i + \varepsilon.
\end{cases}
\]
\begin{proof}
Supongamos que $i = \inf A$.

(i) Como $i$ es el ínfimo de $A$, por definición $i$ es cota inferior de $A$.
Es decir, para todo $a \in A$ se cumple $i \le a$.

(ii) Sea $\varepsilon > 0$. Supongamos, buscando una contradicción, que no existe
$a \in A$ tal que $i \le a < i + \varepsilon$. Entonces, para todo $a \in A$
se cumple $a \ge i + \varepsilon$, de modo que $i + \varepsilon$ es una cota
inferior de $A$. Como además $\varepsilon>0$, tenemos $i + \varepsilon > i$,
lo que contradice que $i$ es la mayor de las cotas inferiores de $A$.
Por lo tanto, debe existir $a \in A$ tal que $i \le a < i + \varepsilon$.

Recíprocamente, supongamos que se verifican:
(i) $i \le a$ para todo $a \in A$;
(ii) para todo $\varepsilon>0$ existe $a \in A$ tal que $i \le a < i + \varepsilon$.

De (i) se sigue que $i$ es cota inferior de $A$.
Sea $i'$ otra cota inferior de $A$. Queremos ver que $i' \le i$.
Supongamos, buscando una contradicción, que $i' > i$.
Sea $\varepsilon = i' - i > 0$. Por (ii) existe $a \in A$ tal que
$i \le a < i + \varepsilon$. Como $i + \varepsilon = i'$, obtenemos
$a < i'$, lo cual contradice que $i'$ es cota inferior de $A$.
Por lo tanto $i' \le i$ y, en consecuencia, $i = \inf A$.
\end{proof}


\subsection{Supremo}
Sea $A \subset\mathbb{R}$, no vacío y acotado superiormente.
\[
s = \sup A \Longleftrightarrow
\begin{cases}
s \ge a & \text{para todo } a \in A,\\[4pt]
\forall \varepsilon > 0 \ \exists a \in A \text{ tal que } s - \varepsilon < a \le s.
\end{cases}
\]
\begin{proof}
Supongamos que $s = \sup A$.

(i) Como $s$ es el supremo de $A$, por definición $s$ es cota superior de $A$.
Es decir, para todo $a \in A$ se cumple $a \le s$.

(ii) Sea $\varepsilon>0$. Supongamos, buscando una contradicción, que no existe
$a \in A$ tal que $s-\varepsilon < a \le s$. Entonces, para todo $a \in A$
se cumple $a \le s-\varepsilon$, y por lo tanto $s-\varepsilon$ es una cota
superior de $A$. Como además $\varepsilon>0$, tenemos $s-\varepsilon < s$,
lo que contradice que $s$ es la menor de las cotas superiores de $A$.
Por lo tanto, debe existir $a \in A$ tal que $s-\varepsilon < a \le s$.

Supongamos que se cumplen:
\\
(i) $a \le s$ para todo $a \in A$\\
(ii) para todo $\varepsilon>0$ existe $a \in A$ tal que $s-\varepsilon < a \le s$.

Entonces (i) dice que $s$ es cota superior de $A$.
Sea ahora $s'$ otra cota superior de $A$. Queremos ver que $s \le s'$.
Supongamos, buscando una contradicción, que $s' < s$.
Sea $\varepsilon = s - s' > 0$. Por (ii) existe $a \in A$ tal que
$s-\varepsilon < a \le s$. Como $s-\varepsilon = s'$, obtenemos $s' < a$,
lo cual contradice que $s'$ es cota superior de $A$.
Por lo tanto $s \le s'$ y, en consecuencia, $s = \sup A$.
\end{proof}

\subsection{Principio de Arquímedes}

\subsubsection*{Versión 1}
\[
\forall x \in \R \ \exists n \in \N \text{ tal que } x \le n.
\]

\begin{proof}[Demostración]
Supongamos por el absurdo que el conjunto de los naturales $\N$ está acotado superiormente.
Como $\N$ es no vacío, por el axioma de completitud existe $s = \sup \N$.

Tomamos $\varepsilon = 1$. Por la propiedad caracterizadora del supremo, existe
$n \in \N$ tal que
\[
s - 1 < n \le s.
\]
De $s - 1 < n$ se sigue que $n + 1 > s$. Como $n \in \N$, también $n+1 \in \N$,
y por lo tanto hemos encontrado un número natural estrictamente mayor que $s$,
lo que contradice que $s$ sea cota superior de $\N$.

Esta contradicción muestra que $\N$ no está acotado superiormente, es decir,
para todo $M \in \R$ existe $n \in \N$ tal que $n > M$. En particular, dado
$x \in \R$, tomando $M = x$ obtenemos un $n \in \N$ con $n \ge x$, que es
justamente lo que afirma la versión 1.
\end{proof}


\subsubsection*{Versión 2}
\[
\forall y > 0 \ \exists n \in \N \text{ tal que } 0 < \frac{1}{n} < y.
\]

\begin{proof}[Demostración]
Sea $y > 0$. Por la Versión 1 del principio de Arquímedes aplicada a $x = 1/y$,
existe $n \in \N$ tal que
\[
n > \frac{1}{y}.
\]
Como $n > 0$, al invertir la desigualdad obtenemos
\[
0 < \frac{1}{n} < y.
\]
Por lo tanto, para todo $y > 0$ existe $n \in \N$ tal que $0 < \dfrac{1}{n} < y$,
como queríamos demostrar.
\end{proof}

\subsection{Densidad de $\Q$}
\[
\forall x, y \in \R \text{ con } x < y \ \exists q \in \Q \text{ tal que } x < q < y.
\]
\begin{proof}[Demostración]
Sean $x, y \in \R$ con $x < y$. Entonces $y - x > 0$ y, por la Versión 2 del
Principio de Arquímedes, existe $n \in \N$ tal que
\[
0 < \frac{1}{n} < y - x,
\]
es decir, $n(y-x) > 1$, o equivalentemente,
\[
ny - nx > 1.
\]

Veamos que existe un entero $m$ tal que $nx < m < ny$. Sea
\[
S = \{ j \in \Z : j > nx \}.
\]
Por la Versión 1 del Principio de Arquímedes, existe un natural mayor que $nx$,
así que $S \neq \emptyset$; además $S$ está acotado inferiormente por $nx$. Como
todo subconjunto no vacío de $\Z$ acotado inferiormente tiene mínimo, sea
$m = \min S$. Por definición de $S$, $m > nx$. Como $m$ es el mínimo de $S$,
$m - 1 \notin S$, es decir $m - 1 \le nx$, de donde
\[
m \le nx + 1 < ny,
\]
usando que $ny - nx > 1$. Por lo tanto, $nx < m < ny$.

Dividiendo por $n > 0$ obtenemos
\[
x < \frac{m}{n} < y.
\]
Tomando $q = \dfrac{m}{n} \in \Q$, concluimos que $x < q < y$, como queríamos
demostrar.
\end{proof}

\subsection{Sucesiones}

\begin{defi}[Sucesión]
Una sucesión de números reales es una función $a : \N \to \R$. Para cada $n \in \N$
notamos $a_n = a(n)$, y a la sucesión completa la notamos $(a_n)_{n\in\N}$.
\end{defi}

\begin{defi}[Límite de una sucesión]
Sea $(a_n)_{n\in\N}$ una sucesión real y $l \in \R$. Decimos que $(a_n)$ converge a
$l$, y notamos $\lim_{n\to\infty} a_n = l$ o $a_n \to l$, si
\[
\forall \varepsilon > 0 \ \exists n_0 \in \N \ \text{tal que} \ \forall n \ge n_0:
\ |a_n - l| < \varepsilon.
\]
\end{defi}

\begin{prop}[Unicidad del límite]
Sea $(a_n)_{n\in\N}$ una sucesión real. Supongamos que $a_n \to l_1$ y $a_n \to l_2$
con $l_1, l_2 \in \R$. Entonces $l_1 = l_2$.
\end{prop}

\begin{proof}
Sea $\varepsilon > 0$. Como $a_n \to l_1$, existe $n_1 \in \N$ tal que
$|a_n - l_1| < \tfrac{\varepsilon}{2}$ para todo $n \ge n_1$. Como $a_n \to l_2$,
existe $n_2 \in \N$ tal que $|a_n - l_2| < \tfrac{\varepsilon}{2}$ para todo
$n \ge n_2$. Tomando $n = \max\{n_1, n_2\}$ y usando la desigualdad triangular,
\[
|l_1 - l_2| = |l_1 - a_n + a_n - l_2| \le |l_1 - a_n| + |a_n - l_2|
< \frac{\varepsilon}{2} + \frac{\varepsilon}{2} = \varepsilon.
\]
Como esto vale para todo $\varepsilon > 0$, no puede ser $|l_1 - l_2| > 0$: si lo
fuera, tomando $\varepsilon = |l_1 - l_2|$ obtendríamos $|l_1 - l_2| < |l_1 - l_2|$,
lo cual es absurdo. Por lo tanto $|l_1 - l_2| = 0$, es decir, $l_1 = l_2$.
\end{proof}

\begin{defi}[Sucesión acotada]
Una sucesión $(a_n)_{n\in\N}$ está acotada si el conjunto $\{a_n : n \in \N\}$ está
acotado, es decir, si existe $M > 0$ tal que $|a_n| \le M$ para todo $n \in \N$.
\end{defi}

\begin{prop}[Sucesión convergente es acotada]
Sea $(a_n)_{n\in\N}$ una sucesión real y sea $l \in \R$ tal que
\[
\lim_{n\to\infty} a_n = l.
\]
Entonces la sucesión $(a_n)$ está acotada.
\end{prop}

\begin{proof}
Por hipótesis, $\lim_{n\to\infty} a_n = l$. Esto significa que
\[
\forall \varepsilon > 0 \ \exists n_0 \in \N \ \text{tal que} \ \forall n \ge n_0:
\ |a_n - l| < \varepsilon.
\]

Tomamos ahora $\varepsilon = 1$. Entonces existe $n_0 \in \N$ tal que para todo
$n \ge n_0$ se cumple
\[
|a_n - l| < 1.
\]
Por la desigualdad triangular,
\[
|a_n| = |(a_n - l) + l| \le |a_n - l| + |l| < 1 + |l|
\]
para todo $n \ge n_0$. Es decir, para todos los índices grandes,
\[
|a_n| \le |l| + 1.
\]

Consideremos ahora los primeros términos de la sucesión:
\[
a_1, a_2, \dots, a_{n_0-1}.
\]
Se trata de un conjunto finito de números reales, por lo que el conjunto
\[
\{|a_1|, |a_2|, \dots, |a_{n_0-1}|\}
\]
tiene un máximo. Sea
\[
M_0 = \max\{|a_1|, |a_2|, \dots, |a_{n_0-1}|\}
\]
(en el caso $n_0 = 1$ podemos tomar, por ejemplo, $M_0 = 0$).

Definimos ahora
\[
M = \max\{M_0,\ |l| + 1\}.
\]

Entonces:
\\- Si $n < n_0$, se cumple $|a_n| \le M_0 \le M$.
\\- Si $n \ge n_0$, se cumple $|a_n| \le |l| + 1 \le M$.
\\
En ambos casos obtenemos
\[
|a_n| \le M \quad \text{para todo } n \in \N.
\]
Por lo tanto, la sucesión $(a_n)$ está acotada.
\end{proof}

\begin{prop}[Álgebra de límites]
Sean $(a_n)_{n\in\N}$ y $(b_n)_{n\in\N}$ sucesiones reales tales que
$\lim_{n\to\infty} a_n = a$ y $\lim_{n\to\infty} b_n = b$, con $a, b \in \R$.
Entonces:
\begin{enumerate}[label=(\roman*)]
\item $\displaystyle\lim_{n\to\infty} (c \cdot a_n) = c \cdot a$ para todo $c \in \R$.
\item $\displaystyle\lim_{n\to\infty} (a_n + b_n) = a + b$.
\item $\displaystyle\lim_{n\to\infty} (a_n b_n) = ab$.
\item Si $b \neq 0$, entonces $\displaystyle\lim_{n\to\infty} \left(\frac{a_n}{b_n}\right) = \frac{a}{b}$.
\item Si existe $n_0 \in \N$ tal que $a_n \le b_n$ para todo $n \ge n_0$, entonces $a \le b$.
\end{enumerate}
\end{prop}

\begin{proof}
Probamos cada ítem por separado.

\medskip
\noindent\textbf{(i)} Si $c = 0$, entonces $c \cdot a_n = 0 = c \cdot a$ para todo
$n$, y la propiedad se cumple trivialmente. Supongamos $c \neq 0$ y sea
$\varepsilon > 0$. Como $a_n \to a$, existe $n_0 \in \N$ tal que
$|a_n - a| < \tfrac{\varepsilon}{|c|}$ para todo $n \ge n_0$. Entonces
\[
|c a_n - c a| = |c||a_n - a| < |c| \cdot \frac{\varepsilon}{|c|} = \varepsilon
\]
para todo $n \ge n_0$.

\medskip
\noindent\textbf{(ii)} Sea $\varepsilon > 0$. Como $a_n \to a$, existe $n_1 \in \N$
tal que $|a_n - a| < \tfrac{\varepsilon}{2}$ para todo $n \ge n_1$. Como
$b_n \to b$, existe $n_2 \in \N$ tal que $|b_n - b| < \tfrac{\varepsilon}{2}$ para
todo $n \ge n_2$. Sea $n_0 = \max\{n_1, n_2\}$. Entonces, para todo $n \ge n_0$,
\[
|(a_n + b_n) - (a+b)| \le |a_n - a| + |b_n - b| < \frac{\varepsilon}{2} +
\frac{\varepsilon}{2} = \varepsilon.
\]

\medskip
\noindent\textbf{(iii)} Como $(a_n)$ converge, por la proposición anterior está
acotada: existe $M > 0$ tal que $|a_n| \le M$ para todo $n \in \N$. Sea
$\varepsilon > 0$. Como $b_n \to b$, existe $n_1 \in \N$ tal que
$|b_n - b| < \tfrac{\varepsilon}{2M}$ para todo $n \ge n_1$. Como $a_n \to a$,
existe $n_2 \in \N$ tal que $|a_n - a| < \tfrac{\varepsilon}{2(|b|+1)}$ para todo
$n \ge n_2$. Sea $n_0 = \max\{n_1, n_2\}$. Entonces, para todo $n \ge n_0$,
\[
|a_n b_n - ab| = |a_n b_n - a_n b + a_n b - ab|
\le |a_n||b_n - b| + |b||a_n - a|
\le M|b_n - b| + |b||a_n - a|.
\]
Acotando cada término,
\[
M|b_n - b| < M \cdot \frac{\varepsilon}{2M} = \frac{\varepsilon}{2},
\qquad
|b||a_n - a| < |b| \cdot \frac{\varepsilon}{2(|b|+1)} < \frac{\varepsilon}{2},
\]
donde en la última desigualdad usamos $\frac{|b|}{|b|+1} < 1$. Sumando,
\[
|a_n b_n - ab| < \frac{\varepsilon}{2} + \frac{\varepsilon}{2} = \varepsilon
\]
para todo $n \ge n_0$.

\medskip
\noindent\textbf{(iv)} Por el ítem (iii), alcanza con probar que
$\dfrac{1}{b_n} \to \dfrac{1}{b}$, ya que entonces
$\dfrac{a_n}{b_n} = a_n \cdot \dfrac{1}{b_n} \to a \cdot \dfrac{1}{b} = \dfrac{a}{b}$.

Por el lema anterior, existe $n_1 \in \N$ tal que $|b_n| \ge \tfrac{|b|}{2}$ para
todo $n \ge n_1$. Sea $\varepsilon > 0$. Como $b_n \to b$, existe $n_2 \in \N$ tal
que $|b_n - b| < \tfrac{\varepsilon |b|^2}{2}$ para todo $n \ge n_2$. Sea
$n_0 = \max\{n_1, n_2\}$. Entonces, para todo $n \ge n_0$,
\[
\left| \frac{1}{b_n} - \frac{1}{b} \right|
= \frac{|b - b_n|}{|b_n||b|}
\le \frac{|b_n - b|}{\tfrac{|b|}{2} \cdot |b|}
= \frac{2|b_n - b|}{|b|^2}
< \frac{2}{|b|^2} \cdot \frac{\varepsilon |b|^2}{2}
= \varepsilon.
\]

\medskip
\noindent\textbf{(v)} Supongamos, buscando una contradicción, que $a > b$. Sea
$\varepsilon = \tfrac{a-b}{2} > 0$. Como $a_n \to a$, existe $n_1 \in \N$ tal que
$a_n > a - \varepsilon$ para todo $n \ge n_1$. Como $b_n \to b$, existe
$n_2 \in \N$ tal que $b_n < b + \varepsilon$ para todo $n \ge n_2$. Sea
$n' = \max\{n_0, n_1, n_2\}$, donde $n_0$ es el índice de la hipótesis
$a_n \le b_n$. Entonces, para todo $n \ge n'$,
\[
a_n > a - \varepsilon = a - \frac{a-b}{2} = \frac{a+b}{2}
= b + \frac{a-b}{2} = b + \varepsilon > b_n,
\]
lo cual contradice que $a_n \le b_n$ para todo $n \ge n_0$. Por lo tanto,
necesariamente $a \le b$.
\end{proof}

\begin{defi}[Sucesión monótona]
Una sucesión $(a_n)_{n\in\N}$ es monótona creciente si $a_n \le a_{n+1}$ para todo
$n \in \N$, y es monótona decreciente si $a_n \ge a_{n+1}$ para todo $n \in \N$.
\end{defi}

\begin{prop}
Sea $(a_n)_{n\in\N}$ una sucesión real monótona creciente y acotada superiormente.
Sea
\[
s = \sup\{a_n : n \in \N\}.
\]
Entonces
\[
\lim_{n\to\infty} a_n = s.
\]
\end{prop}

\begin{proof}
Como $(a_n)$ está acotada superiormente, el conjunto
\[
A = \{a_n : n \in \N\}
\]
tiene supremo $s \in \R$.

Queremos probar que $\lim_{n\to\infty} a_n = s$, es decir,
\[
\forall \varepsilon > 0 \ \exists n_0 \in \N \ \forall n \ge n_0:
\ |a_n - s| < \varepsilon.
\]

Sea entonces $\varepsilon > 0$ arbitrario. Por la propiedad
caracterizadora del supremo aplicada al conjunto $A$, existe
$n_0 \in \N$ tal que
\[
s - \varepsilon < a_{n_0} \le s.
\]

Como la sucesión $(a_n)$ es monótona creciente, se cumple
\[
a_{n_0} \le a_n \le s \quad \text{para todo } n \ge n_0.
\]

De $a_n \le s$ obtenemos $a_n - s \le 0$, luego
\[
|a_n - s| = s - a_n.
\]
Además, de $a_{n_0} \le a_n$ se sigue
\[
s - a_n \le s - a_{n_0}.
\]
Juntando estas desigualdades,
\[
|a_n - s|
= s - a_n
\le s - a_{n_0}
< s - (s - \varepsilon)
= \varepsilon
\]
para todo $n \ge n_0$.

Por lo tanto, para todo $\varepsilon>0$ encontramos $n_0 \in \N$ tal que
para todo $n \ge n_0$ se cumple $|a_n - s|<\varepsilon$, y esto significa
exactamente que $\lim_{n\to\infty} a_n = s$.
\end{proof}

\begin{prop}
Sea $(a_n)_{n\in\N}$ una sucesión real monótona decreciente y acotada inferiormente.
Sea
\[
i = \inf\{a_n : n \in \N\}.
\]
Entonces
\[
\lim_{n\to\infty} a_n = i.
\]
\end{prop}

\begin{proof}
Como $(a_n)$ está acotada inferiormente, el conjunto
\[
A = \{a_n : n \in \N\}
\]
tiene ínfimo $i \in \R$.

Queremos probar que $\lim_{n\to\infty} a_n = i$, es decir,
\[
\forall \varepsilon > 0 \ \exists n_0 \in \N \ \forall n \ge n_0:
\ |a_n - i| < \varepsilon.
\]

Sea $\varepsilon>0$ arbitrario. Por la propiedad caracterizadora del
ínfimo aplicada al conjunto $A$, existe $n_0 \in \N$ tal que
\[
i \le a_{n_0} < i + \varepsilon.
\]

Como la sucesión $(a_n)$ es monótona decreciente, se cumple
\[
i \le a_n \le a_{n_0} \quad \text{para todo } n \ge n_0.
\]

De $a_n \ge i$ se obtiene $a_n - i \ge 0$, luego
\[
|a_n - i| = a_n - i.
\]
Además, de $a_n \le a_{n_0}$ se sigue
\[
a_n - i \le a_{n_0} - i.
\]
Por lo tanto,
\[
|a_n - i|
= a_n - i
\le a_{n_0} - i
< (i + \varepsilon) - i
= \varepsilon
\]
para todo $n \ge n_0$.

Entonces, para todo $\varepsilon>0$ encontramos $n_0 \in \N$ tal que
para todo $n \ge n_0$ se cumple $|a_n - i|<\varepsilon$, lo cual prueba
que $\lim_{n\to\infty} a_n = i$.
\end{proof}

\begin{prop}[Equivalencia del supremo]
Sea $A \subset \R$ un conjunto no vacío y acotado superiormente, y sea $s \in \R$.
Entonces
\[
s = \sup A
\]
si y sólo si se cumplen:
\begin{enumerate}[label=(\roman*)]
    \item $s$ es cota superior de $A$, es decir, $a \le s$ para todo $a \in A$;
    \item existe una sucesión $(a_n)_{n\in\N}$ con $a_n \in A$ para todo $n$, tal que
    \[
    \lim_{n\to\infty} a_n = s.
    \]
\end{enumerate}
\end{prop}

\begin{proof}
Supongamos primero que $s = \sup A$. Entonces, por definición de supremo,
$s$ es cota superior de $A$, con lo cual se cumple (i).

Nos queda probar (ii). Usamos la caracterización del supremo que ya vimos:
para todo $\varepsilon > 0$ existe $a \in A$ tal que
\[
s - \varepsilon < a \le s.
\]
Para cada $n \in \N$, aplicamos esta propiedad con $\varepsilon = \tfrac{1}{n}$.
Obtenemos así un elemento $a_n \in A$ tal que
\[
s - \frac{1}{n} < a_n \le s.
\]
Esto define una sucesión $(a_n)_{n\in\N}$ de elementos de $A$.

Veamos ahora que $\lim_{n\to\infty} a_n = s$. Sea $\varepsilon > 0$. Por la
Versión 2 del Principio de Arquímedes, existe $n_0 \in \N$ tal que
$\frac{1}{n_0} < \varepsilon$. Entonces, si $n \ge n_0$,
se tiene $\frac{1}{n} \le \frac{1}{n_0} < \varepsilon$, y por la construcción
de $(a_n)$ se cumple
\[
s - \frac{1}{n} < a_n \le s.
\]
De aquí, $s - \varepsilon = s - \frac{1}{n} < a_n \le s$, luego
\[
0 \le s - a_n < \varepsilon,
\]
lo que implica
\[
|a_n - s| = s - a_n < \varepsilon.
\]
Por lo tanto,
\[
\forall \varepsilon > 0\ \exists n_0 \in \N\ \forall n \ge n_0:\ |a_n - s| < \varepsilon,
\]
es decir, $\lim_{n\to\infty} a_n = s$. Esto prueba (ii).

\medskip

Recíprocamente, supongamos que se cumplen (i) y (ii). Entonces $s$ es cota
superior de $A$. Para ver que $s = \sup A$, basta probar que $s$ es la menor
de las cotas superiores de $A$. Sea $s'$ otra cota superior de $A$. Queremos
ver que $s \le s'$.

Supongamos, buscando una contradicción, que $s > s'$. Definimos
\[
\varepsilon = s - s' > 0.
\]
Como $\lim_{n\to\infty} a_n = s$, existe $n_0 \in \N$ tal que, para todo
$n \ge n_0$, se cumple
\[
|a_n - s| < \varepsilon.
\]
En particular, para $n \ge n_0$ tenemos
\[
a_n > s - \varepsilon = s - (s - s') = s'.
\]
Pero $a_n \in A$ y $s'$ es cota superior de $A$, luego debería cumplirse
$a_n \le s'$, lo que contradice la desigualdad anterior.

Esta contradicción muestra que no puede ocurrir $s > s'$, por lo que necesariamente
$s \le s'$. Como $s'$ era una cota superior cualquiera de $A$, concluimos que
$s$ es la menor de las cotas superiores de $A$, es decir, $s = \sup A$.

Queda así demostrada la equivalencia.
\end{proof}

\begin{defi}[Subsucesión]
Sea $(a_n)_{n\in\N}$ una sucesión real. Una subsucesión de $(a_n)$ es una sucesión
de la forma $(a_{n_j})_{j\in\N}$, donde $(n_j)_{j\in\N}$ es una sucesión
estrictamente creciente de números naturales, es decir, $n_1 < n_2 < n_3 < \dots$.
\end{defi}

\begin{prop}
Sea $(a_n)_{n\in\N}$ una sucesión real y sea $(a_{n_j})_{j\in\N}$ una subsucesión de $(a_n)$,
donde $(n_j)_{j\in\N}$ es una sucesión estrictamente creciente de números naturales.
Si
\[
\lim_{n\to\infty} a_n = l,
\]
entonces
\[
\lim_{j\to\infty} a_{n_j} = l.
\]
\end{prop}

\begin{proof}
Supongamos que $\lim_{n\to\infty} a_n = l$. Por definición de límite, esto significa que
\[
\forall \varepsilon > 0 \ \exists n_0 \in \N \ \text{tal que} \ \forall n \ge n_0:
\ |a_n - l| < \varepsilon.
\]

Sea ahora $\varepsilon > 0$ fijo. Por la hipótesis anterior, existe $n_0 \in \N$ tal que
\[
n \ge n_0 \ \Rightarrow\ |a_n - l| < \varepsilon.
\]

Como $(n_j)_{j\in\N}$ es una sucesión estrictamente creciente de números naturales,
podemos elegir $j_0 \in \N$ tal que
\[
n_{j_0} \ge n_0.
\]
Entonces, para todo $j \ge j_0$ se tiene
\[
n_j \ge n_{j_0} \ge n_0.
\]
Aplicando la propiedad de arriba a $n = n_j$, obtenemos
\[
|a_{n_j} - l| < \varepsilon \quad \text{para todo } j \ge j_0.
\]

Por lo tanto,
\[
\forall \varepsilon > 0 \ \exists j_0 \in \N \ \forall j \ge j_0:\ |a_{n_j} - l| < \varepsilon,
\]
lo cual significa exactamente que
\[
\lim_{j\to\infty} a_{n_j} = l.
\]
\end{proof}
