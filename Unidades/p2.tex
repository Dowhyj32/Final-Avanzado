\subsection{Conjuntos coordinables y cardinal}

\begin{defi}
Sean $X$ e $Y$ dos conjuntos. Decimos que son \emph{coordinables}
(o \emph{equipotentes}, o que tienen el mismo cardinal) si existe
una funciĂłn biyectiva $f : X \to Y$. En este caso escribimos
\[
X \sim Y.
\]
\end{defi}

\begin{prop}
La relaciĂłn $\sim$ es una relaciĂłn de equivalencia en la clase de
todos los conjuntos.
\end{prop}

\begin{proof}
Debemos probar que la relaciĂłn $\sim$ es reflexiva, simĂ©trica
y transitiva.

\medskip

\noindent\textbf{Reflexividad.}
Sea $X$ un conjunto cualquiera. Consideramos la funciĂłn identidad
\[
\operatorname{id}_X : X \to X, \quad \operatorname{id}_X(x) = x.
\]
La funciĂłn identidad es inyectiva (si $\operatorname{id}_X(x)
= \operatorname{id}_X(y)$, entonces $x = y$) y sobreyectiva
(para todo $x \in X$ existe $x \in X$ tal que
$\operatorname{id}_X(x) = x$). Luego es biyectiva, y por la
definiciĂłn de $\sim$ se tiene $X \sim X$. Por lo tanto, $\sim$
es reflexiva.

\medskip

\noindent\textbf{SimetrĂ­a.}
Sean $X$ e $Y$ conjuntos tales que $X \sim Y$. Por definiciĂłn,
existe una biyecciĂłn $f : X \to Y$. Toda funciĂłn biyectiva tiene
inversa $f^{-1} : Y \to X$, y dicha inversa es tambiĂ©n biyectiva.
Por lo tanto, existe una biyecciĂłn de $Y$ en $X$, es decir,
$Y \sim X$. Luego, $\sim$ es simĂ©trica.

\medskip

\noindent\textbf{Transitividad.}
Sean $X, Y, Z$ conjuntos tales que $X \sim Y$ y $Y \sim Z$.
Entonces existen biyecciones
\[
f : X \to Y,
\qquad
g : Y \to Z.
\]
Consideramos la composiciĂłn
\[
g \circ f : X \to Z, \quad (g \circ f)(x) = g(f(x)).
\]
Como composiciĂłn de funciones biyectivas, $g \circ f$ es tambiĂ©n
biyectiva: la composiciĂłn de funciones inyectivas es inyectiva y
la composiciĂłn de funciones sobreyectivas es sobreyectiva. En
consecuencia, existe una biyecciĂłn de $X$ en $Z$, es decir,
$X \sim Z$. Esto muestra que $\sim$ es transitiva.

\medskip

Como $\sim$ es reflexiva, simĂ©trica y transitiva, concluimos que
$\sim$ es una relaciĂłn de equivalencia.
\end{proof}

\begin{defi}
Definimos el \emph{cardinal} de un conjunto $X$ como la clase de
equivalencia de los conjuntos coordinables con $X$:
\[
\#X = \operatorname{Card}(X)
:= \{ Y \mid X \sim Y \}.
\]
A algunos cardinales les damos nombres especiales:
\begin{itemize}
    \item $\#\N = \aleph_0$ (cardinal numerable),
    \item $\#\R = \mathfrak{c}$ (el \emph{continuo}),
    \item $\#\{1,2,\dots,n\} = n$ para $n \in \N$.
\end{itemize}
\end{defi}

\begin{defi}
Para $n \in \N$, llamamos
\[
I_n = \{1,2,\dots,n\}
\]
al \emph{intervalo inicial} del conjunto $\N$ de los nĂşmeros naturales.
\end{defi}

\begin{defi}
Un conjunto $A$ es \emph{finito} si existe $n \in \N$ tal que
\[
A \sim I_n.
\]
\end{defi}

\begin{defi}
Un conjunto $A$ es \emph{infinito} si no es finito.
\end{defi}

\begin{defi}
Un conjunto $A$ es \emph{numerable} si $A \sim \N$.
Equivalente y simbĂłlicamente, si
\[
\#A = \aleph_0.
\]
\end{defi}
\begin{defi}
Decimos que un conjunto $A$ es \emph{a lo sumo numerable}
(o \emph{contable}) si es finito o numerable. Es decir,
$A$ es a lo sumo numerable si cumple
\[
A \sim I_n \quad \text{para algĂşn } n \in \N
\quad \text{o bien} \quad
A \sim \N.
\]
\end{defi}

\begin{prop}
Sea $A$ un conjunto numerable y sea $B \subseteq A$.
Entonces $B$ es a lo sumo numerable.
\end{prop}

\begin{proof}
Si $B$ es finito, por definiciĂłn ya es a lo sumo numerable y no hay nada que
probar. Supongamos entonces que $B$ es infinito. Veremos que en ese caso
$B$ es numerable.

Como $A$ es numerable, por definiciĂłn existe una biyecciĂłn
\[
f : \N \longrightarrow A.
\]
Consideremos la sucesiĂłn $(f(1), f(2), f(3),\dots)$ de elementos de $A$ y
vamos a â€śextraerâ€ť de ella una enumeraciĂłn de los elementos de $B$.

Definimos, por inducciĂłn, una sucesiĂłn estrictamente creciente
$(n_k)_{k\in\N}$ de nĂşmeros naturales de la siguiente manera.

En primer lugar, como $B$ es infinito, en particular es no vacĂ­o y existe
algĂşn $b_1 \in B$. Como $f$ es sobreyectiva, existe $n_1 \in \N$ tal que
$f(n_1) = b_1$. AdemĂˇs, podemos elegir $n_1$ como el mĂ­nimo de los
naturales $n$ que satisfacen $f(n) \in B$:
\[
n_1 = \min\{\,n \in \N : f(n) \in B\,\}.
\]
Este mĂ­nimo existe porque el conjunto entre llaves es no vacĂ­o y estĂˇ
contenido en $\N$.

Supuesto definido $n_k$ para algĂşn $k \in \N$, definimos $n_{k+1}$ asĂ­.
Como $B$ es infinito, el conjunto
\[
B_k := B \setminus \{f(n_1), f(n_2), \dots, f(n_k)\}
\]
no es vacĂ­o (si fuera vacĂ­o, $B$ tendrĂ­a a lo sumo $k$ elementos y serĂ­a
finito). Entonces existe $b_{k+1} \in B_k$. Nuevamente, como $f$ es
sobreyectiva, existe $m \in \N$ tal que $f(m) = b_{k+1}$. AdemĂˇs, podemos
elegir $m$ de manera que $m > n_k$ (basta tomar algĂşn Ă­ndice de $b_{k+1}$
mayor que todos $n_1,\dots,n_k$). Definimos
\[
n_{k+1} = \min\{\,n \in \N : n > n_k,\ f(n) \in B_k\,\}.
\]
Este mĂ­nimo existe porque el conjunto entre llaves es no vacĂ­o y contenido
en los naturales mayores que $n_k$. De la definiciĂłn se deduce que
\[
n_1 < n_2 < n_3 < \dots,
\]
es decir, $(n_k)$ es estrictamente creciente.

Definimos ahora una funciĂłn
\[
g : \N \longrightarrow B, \qquad g(k) = f(n_k).
\]

Veamos que $g$ es biyectiva.

\medskip

\noindent\textbf{Inyectividad.}
Sean $k,\ell \in \N$ tales que $g(k) = g(\ell)$, es decir,
$f(n_k) = f(n_\ell)$. Como $f$ es inyectiva, se sigue que $n_k = n_\ell$.
Pero la sucesiĂłn $(n_k)$ es estrictamente creciente, luego de
$n_k = n_\ell$ se deduce $k = \ell$. Por lo tanto, $g$ es inyectiva.

\medskip

\noindent\textbf{Sobreyectividad.}
Sea $b \in B$. Como $f$ es sobreyectiva, existe $n \in \N$ tal que
$f(n) = b$. En el proceso de construcciĂłn de la sucesiĂłn $(n_k)$, en algĂşn
paso $k$ el elemento $b$ aparece por primera vez entre los valores de $f$;
es decir, existe un Ăşnico $k$ tal que $n_k$ es el mĂ­nimo Ă­ndice con
$f(n_k) = b$ y $n_k > n_{k-1}$ (para $k=1$ entendemos que no hay condiciĂłn
anterior). Por la definiciĂłn de $g$, se tiene entonces
\[
g(k) = f(n_k) = b.
\]
De este modo, para todo $b \in B$ existe $k \in \N$ tal que $g(k) = b$, y
por lo tanto $g$ es sobreyectiva.

\medskip

Hemos construido una biyecciĂłn $g : \N \to B$, lo cual muestra que $B$ es
numerable. Recordando que al principio separamos el caso en que $B$ es
finito, concluimos que, en todos los casos, $B$ es a lo sumo numerable.
\end{proof}

\begin{teo}
Sea $A$ un conjunto infinito. Entonces existe un subconjunto
$B \subseteq A$ tal que $B$ es numerable.
\end{teo}

\begin{proof}
Como $A$ es infinito, en particular es no vacĂ­o, de modo que podemos elegir
un elemento $a_1 \in A$.

Supondremos construidos elementos distintos $a_1,\dots,a_n \in A$ para algĂşn
$n \in \N$. Consideremos el conjunto
\[
F_n = \{a_1,\dots,a_n\}.
\]
Si $A \setminus F_n$ fuera vacĂ­o, tendrĂ­amos $A = F_n$, es decir, $A$
serĂ­a finito, lo cual contradice la hipĂłtesis de que $A$ es infinito.
Por lo tanto,
\[
A \setminus F_n \neq \varnothing,
\]
y podemos elegir un elemento
\[
a_{n+1} \in A \setminus F_n.
\]
En particular, $a_{n+1} \in A$ y $a_{n+1} \notin \{a_1,\dots,a_n\}$, por
lo que los elementos $a_1,\dots,a_{n+1}$ siguen siendo todos distintos.

De este modo, por inducciĂłn, obtenemos una sucesiĂłn $(a_n)_{n\in\N}$ de
elementos de $A$ tales que
\[
a_n \neq a_m \quad \text{si } n \neq m.
\]

Definimos ahora
\[
B = \{a_n : n \in \N\}.
\]
Claramente $B \subseteq A$, por construcciĂłn.

Consideremos la funciĂłn
\[
f : \N \longrightarrow B, \qquad f(n) = a_n.
\]

\emph{Inyectividad.}  
Si $f(n) = f(m)$, entonces $a_n = a_m$, y como la sucesiĂłn $(a_n)$ tiene
todos sus tĂ©rminos distintos, se sigue que $n = m$. Por lo tanto, $f$ es
inyectiva.

\emph{Sobreyectividad.}  
Sea $b \in B$. Por definiciĂłn de $B$, existe $n \in \N$ tal que $b = a_n$.
Entonces $f(n) = a_n = b$, de modo que todo elemento de $B$ es imagen de
algĂşn $n \in \N$. Por lo tanto, $f$ es sobreyectiva.

Concluimos que $f : \N \to B$ es una biyecciĂłn, es decir, $B$ es numerable.
Como ademĂˇs $B \subseteq A$, hemos encontrado un subconjunto numerable de
$A$, tal como querĂ­amos.
\end{proof}

\subsection{Orden entre cardinales}

Recordemos que, por definiciĂłn,
\[
\#A = \#B \quad \Longleftrightarrow \quad A \sim B
\]
es decir, si y sĂłlo si existe una biyecciĂłn \(f : A \to B\).

\begin{defi}
Sean $X$ e $Y$ conjuntos. Decimos que
\[
\#X \le \#Y
\]
si existe una funciĂłn inyectiva \(f : X \to Y\).
\end{defi}

\begin{defi}
Decimos que
\[
\#X < \#Y
\]
si se cumplen:
\begin{enumerate}[label=(\roman*)]
    \item \(\#X \le \#Y\), es decir, existe una inyecciĂłn \(f : X \to Y\);
    \item \(X \not\sim Y\), es decir, no existe biyecciĂłn entre \(X\) e \(Y\).
\end{enumerate}
\end{defi}

\begin{prop}[Buena definiciĂłn del orden entre cardinales]
Sean $A, B, X, Y$ conjuntos tales que $\#A \le \#B$, $A \sim X$ y $B \sim Y$.
Entonces $\#X \le \#Y$.
\end{prop}

\begin{proof}
Como $\#A \le \#B$, existe una funciĂłn inyectiva $f : A \to B$. Como $A \sim X$
y $B \sim Y$, existen biyecciones $h : A \to X$ y $g : B \to Y$. Como $h$ es
biyectiva, tambiĂ©n lo es su inversa $h^{-1} : X \to A$. Consideremos la
composiciĂłn
\[
g \circ f \circ h^{-1} : X \to Y.
\]
Como $h^{-1}$, $f$ y $g$ son inyectivas, y la composiciĂłn de funciones
inyectivas es inyectiva, concluimos que $g \circ f \circ h^{-1}$ es inyectiva.
Por lo tanto, $\#X \le \#Y$.
\end{proof}

\begin{prop}[Propiedades de orden]
La relaciĂłn $\le$ entre cardinales es reflexiva y transitiva. Es decir:
\begin{enumerate}[label=(\roman*)]
\item $\#A \le \#A$ para todo conjunto $A$.
\item Si $\#A \le \#B$ y $\#B \le \#C$, entonces $\#A \le \#C$.
\end{enumerate}
\end{prop}

\begin{proof}
(i) La funciĂłn identidad $\operatorname{id}_A : A \to A$ es inyectiva, luego
$\#A \le \#A$.

(ii) Supongamos $\#A \le \#B$ y $\#B \le \#C$. Entonces existen funciones
inyectivas $f : A \to B$ y $g : B \to C$. La composiciĂłn $g \circ f : A \to C$
es inyectiva, por ser composiciĂłn de funciones inyectivas. Por lo tanto,
$\#A \le \#C$.
\end{proof}

\begin{obs}
La antisimetrĂ­a de $\le$ (es decir, que $\#A \le \#B$ y $\#B \le \#A$
impliquen $\#A = \#B$) no es inmediata: es el contenido del Teorema de
Cantor--SchrĂ¶der--Bernstein, enunciado mĂˇs adelante.
\end{obs}

\begin{prop}
Sean $X$ e $Y$ conjuntos. Existe una funciĂłn inyectiva
$f : X \to Y$ si y sĂłlo si existe una funciĂłn sobreyectiva
$g : Y \to X$.
\end{prop}

\begin{proof}
\textbf{($\Rightarrow$)} Supongamos que existe una funciĂłn inyectiva
$f : X \to Y$. Distinguimos dos casos.

Si $X = \varnothing$, entonces $f$ es la Ăşnica funciĂłn posible
$\varnothing \to Y$. En este caso, la Ăşnica funciĂłn de $Y$ a $X$
es la funciĂłn vacĂ­a $g : Y \to \varnothing$, que es sobreyectiva
sĂłlo si $Y = \varnothing$. En muchas aplicaciones se descarta el
caso trivial $X = \varnothing$, asĂ­ que supongamos ahora que
$X \ne \varnothing$.

Como $X \ne \varnothing$, elegimos un elemento fijo $x_0 \in X$.
Definimos $g : Y \to X$ de la siguiente manera:
\[
g(y) =
\begin{cases}
x & \text{si existe } x \in X \text{ tal que } f(x) = y,\\
x_0 & \text{si no existe tal } x.
\end{cases}
\]
La inyectividad de $f$ garantiza que, cuando $y$ estĂˇ en la imagen
de $f$, el elemento $x$ tal que $f(x) = y$ es Ăşnico, de modo que
$g$ estĂˇ bien definida.

Veamos que $g$ es sobreyectiva. Sea $x \in X$. Como $f$ es funciĂłn
de $X$ en $Y$, tenemos $f(x) \in Y$, y por definiciĂłn de $g$,
\[
g(f(x)) = x.
\]
Luego todo $x \in X$ es imagen de algĂşn elemento de $Y$ (por ejemplo,
de $f(x)$), y por lo tanto $g$ es sobreyectiva.

\medskip

\textbf{($\Leftarrow$)} RecĂ­procamente, supongamos que existe una funciĂłn
sobreyectiva $g : Y \to X$. Para cada $x \in X$ consideremos el conjunto
de sus preimĂˇgenes:
\[
Y_x = \{\, y \in Y : g(y) = x \,\}.
\]
Como $g$ es sobreyectiva, $Y_x$ es no vacĂ­o para todo $x \in X$.

Elegimos, para cada $x \in X$, un elemento $y_x \in Y_x$ (es decir,
$g(y_x) = x$), y definimos
\[
f : X \to Y, \qquad f(x) = y_x.
\]

Probemos que $f$ es inyectiva. Sean $x_1,x_2 \in X$ tales que
$f(x_1) = f(x_2)$. Entonces
\[
y_{x_1} = y_{x_2}.
\]
Aplicando $g$ a ambos lados obtenemos
\[
g(y_{x_1}) = g(y_{x_2}),
\]
es decir,
\[
x_1 = x_2,
\]
ya que por definiciĂłn de $y_x$ se cumple $g(y_x) = x$. Por lo tanto,
$f$ es inyectiva.

\medskip

Hemos probado en un sentido que de una inyectiva $X \to Y$ obtenemos
una sobreyectiva $Y \to X$, y en el otro que de una sobreyectiva
$Y \to X$ obtenemos una inyectiva $X \to Y$. Esto completa la
demostraciĂłn.
\end{proof}

\subsection{Conjunto de partes y teorema de Cantor}

\begin{defi}
Dado un conjunto $X$, llamamos \emph{conjunto de partes} de $X$ al conjunto
\[
\mathcal{P}(X) = \{ A : A \subseteq X\}.
\]
\end{defi}

\begin{teo}[Cantor]
Sea $X$ un conjunto. Entonces
\[
\#X < \#\mathcal{P}(X).
\]
\end{teo}

\begin{proof}
Recordemos que, por la definiciĂłn de orden entre cardinales,
\[
\#X < \#\mathcal{P}(X)
\quad\Longleftrightarrow\quad
\begin{cases}
\text{existe una inyecciĂłn } f : X \to \mathcal{P}(X),\\
\text{no existe biyecciĂłn entre } X \text{ y } \mathcal{P}(X).
\end{cases}
\]

\textbf{(1) Existe una inyecciĂłn $X \to \mathcal{P}(X)$.}

Definimos
\[
f : X \longrightarrow \mathcal{P}(X), \qquad f(x) = \{x\}.
\]
Claramente $f(x) \subseteq X$ para todo $x$, luego $f(x) \in \mathcal{P}(X)$.
Si $f(x) = f(y)$, entonces $\{x\} = \{y\}$ y por lo tanto $x = y$.
AsĂ­, $f$ es inyectiva, y obtenemos
\[
\#X \le \#\mathcal{P}(X).
\]

\textbf{(2) No existe biyecciĂłn entre $X$ y $\mathcal{P}(X)$.}

Basta ver que \emph{no existe ninguna funciĂłn sobreyectiva}
$g : X \to \mathcal{P}(X)$.

Procedemos por absurdo. Supongamos que existe una funciĂłn
\[
g : X \longrightarrow \mathcal{P}(X)
\]
sobreyectiva. Consideremos el subconjunto
\[
B = \{ x \in X : x \notin g(x) \}.
\]
Por definiciĂłn, $B \subseteq X$, de modo que $B \in \mathcal{P}(X)$.

Como $g$ es sobreyectiva, debe existir algĂşn elemento $a \in X$ tal que
\[
g(a) = B.
\]

Estudiemos ahora si $a$ pertenece o no a $B$:

- Supongamos que $a \in B$.  
  Por la definiciĂłn de $B$, esto significa que $a \notin g(a)$. Pero
  $g(a) = B$, luego $a \notin B$, lo que contradice $a \in B$.

- Supongamos que $a \notin B$.  
  Entonces, por la definiciĂłn de $B$, se tiene $a \in g(a)$. Como
  $g(a) = B$, esto implica $a \in B$, contradiciendo $a \notin B$.

En ambos casos llegamos a una contradicciĂłn. Por lo tanto, nuestra
suposiciĂłn inicial es falsa: no existe funciĂłn sobreyectiva
$g : X \to \mathcal{P}(X)$.

Concluimos que no existe biyecciĂłn entre $X$ y $\mathcal{P}(X)$.
Junto con (1), esto implica
\[
\#X < \#\mathcal{P}(X),
\]
como querĂ­amos demostrar.
\end{proof}

\subsection{Suma y resta de conjuntos numerables}

\begin{prop}
Sea $X$ un conjunto infinito. Entonces existe un subconjunto
$Z \subset X$, con $Z$ numerable, tal que
\[
X \sim X \setminus Z.
\]
\end{prop}

\begin{proof}
Como $X$ es infinito, por el teorema anterior existe un subconjunto
numerable infinito $C \subset X$. Como $C$ es numerable, existe una
biyecciĂłn
\[
\varphi : \N \to C.
\]
Escribimos $c_n = \varphi(n)$ para todo $n \in \N$, de modo que
\[
C = \{c_1, c_2, c_3,\dots\}.
\]

Definimos ahora dos subconjuntos disjuntos de $C$:
\[
Z = \{c_{2n} : n \in \N\}, \qquad
D = \{c_{2n-1} : n \in \N\}.
\]
Entonces $C = D \cup Z$ y $D \cap Z = \varnothing$. AdemĂˇs, tanto $D$
como $Z$ son numerables (son imĂˇgenes de $\N$ por las aplicaciones
$n \mapsto c_{2n-1}$ y $n \mapsto c_{2n}$, respectivamente).

Sea
\[
Y = X \setminus C.
\]
Entonces tenemos una particiĂłn
\[
X = Y \cup D \cup Z
\quad\text{(uniĂłn disjunta)}.
\]
Por otra parte,
\[
X \setminus Z = Y \cup D.
\]

Definimos una aplicaciĂłn $f : X \to X \setminus Z$ por:
\[
f(x) =
\begin{cases}
x, & \text{si } x \in Y \cup D,\\
c_{2n-1}, & \text{si } x = c_{2n} \in Z \text{ para algĂşn } n \in \N.
\end{cases}
\]

Veamos que $f$ es biyectiva.

\medskip

\noindent\textbf{Inyectividad.}
- Si $x_1,x_2 \in Y \cup D$ y $f(x_1) = f(x_2)$, entonces $x_1 = x_2$
  porque $f$ actĂşa como la identidad en $Y \cup D$.

- Si $x_1 = c_{2n_1}$ e $x_2 = c_{2n_2}$ pertenecen a $Z$ y
  $f(x_1) = f(x_2)$, entonces
  \[
  c_{2n_1-1} = f(c_{2n_1}) = f(c_{2n_2}) = c_{2n_2-1},
  \]
  de donde $2n_1-1 = 2n_2-1$ y luego $n_1 = n_2$, es decir
  $x_1 = x_2$.

- No puede ocurrir que $x_1 \in Y \cup D$ y $x_2 \in Z$ tengan la
  misma imagen, porque las imĂˇgenes de $Y \cup D$ estĂˇn en $Y \cup D$
  y las de $Z$ estĂˇn contenidas en $D$; pero $Y$ y $D$ son disjuntos.

En todos los casos, de $f(x_1) = f(x_2)$ se deduce $x_1 = x_2$, luego
$f$ es inyectiva.

\medskip

\noindent\textbf{Sobreyectividad.}
Sea $y \in X \setminus Z = Y \cup D$.

- Si $y \in Y$, entonces $f(y) = y$, asĂ­ que $y$ es imagen de sĂ­ mismo.

- Si $y \in D$, digamos $y = c_{2n-1}$ para algĂşn $n \in \N$, entonces
  $f(c_{2n}) = c_{2n-1} = y$, de modo que $y$ es imagen de $c_{2n} \in Z$.

En consecuencia, todo elemento de $X \setminus Z$ es imagen de algĂşn
elemento de $X$, y $f$ es sobreyectiva.

\medskip

Hemos construido una biyecciĂłn $f : X \to X \setminus Z$ con
$Z \subset X$ numerable, por lo que $X \sim X \setminus Z$.
\end{proof}

\begin{prop}
Sea $B$ un conjunto y sea $A$ un conjunto numerable.
Suponemos que $B \setminus A$ es infinito. Entonces
\[
B \sim B \setminus A.
\]
\end{prop}

\begin{proof}
Podemos reemplazar $A$ por $A \cap B$, que sigue siendo numerable y
cumple
\[
B \setminus (A \cap B) = B \setminus A.
\]
Por simplicidad, suponemos desde ahora que $A \subseteq B$.

Como $A$ es numerable, existe una biyecciĂłn
\[
(a_n)_{n\in\N} : \N \to A,
\]
es decir, podemos escribir
\[
A = \{a_1, a_2, a_3,\dots\}.
\]

Por hipĂłtesis, $B \setminus A$ es infinito. Entonces, por el teorema
â€śconjunto infinito contiene un subconjunto numerableâ€ť, existe un
subconjunto numerable infinito
\[

\]
Tomamos una biyecciĂłn
\[
(c_n)_{n\in\N} : \N \to C, \quad C = \{c_1, c_2, c_3,\dots\}.
\]

Definimos ahora una aplicaciĂłn $f : B \to B \setminus A$ por:
\[
f(x) =
\begin{cases}
c_n, & \text{si } x = a_n \in A \text{ para algĂşn } n \in \N,\\[4pt]
x,   & \text{si } x \in B \setminus A.
\end{cases}
\]

Observemos primero que $f$ estĂˇ bien definida: si $x \in A$, entonces
$f(x) = c_n \in C \subseteq B \setminus A$; si $x \in B \setminus A$,
entonces $f(x) = x \in B \setminus A$. En cualquier caso,
$f(x) \in B \setminus A$.

\medskip

\noindent\textbf{Inyectividad.}
- Si $x_1, x_2 \in B \setminus A$ y $f(x_1) = f(x_2)$, entonces
  $x_1 = x_2$ porque $f$ actĂşa como la identidad en $B \setminus A$.

- Si $x_1 = a_n$ y $x_2 = a_m$ pertenecen a $A$ y
  $f(x_1) = f(x_2)$, entonces $c_n = c_m$, y como la sucesiĂłn
  $(c_n)$ tiene todos sus tĂ©rminos distintos, se sigue $n = m$ y
  por lo tanto $x_1 = x_2$.

- No puede ocurrir que $x_1 \in A$ y $x_2 \in B \setminus A$ tengan
  la misma imagen, porque $f(x_1) \in C \subseteq B \setminus A$,
  mientras que $f(x_2) = x_2 \in B \setminus A \setminus C$, y
  $C$ es disjunto de $B \setminus A \setminus C$.

En consecuencia, $f$ es inyectiva.

\medskip

\noindent\textbf{Sobreyectividad.}
Sea $y \in B \setminus A$. Distinguimos dos casos:

- Si $y \in B \setminus (A \cup C)$, entonces $f(y) = y$, de modo que
  $y$ es imagen de sĂ­ mismo.

- Si $y \in C$, digamos $y = c_n$ para algĂşn $n \in \N$, entonces
  $f(a_n) = c_n = y$, de modo que $y$ es imagen de $a_n \in A$.

Por lo tanto, todo elemento de $B \setminus A$ es imagen de algĂşn
elemento de $B$, y $f$ es sobreyectiva.

\medskip

Hemos construido una biyecciĂłn $f : B \to B \setminus A$, lo cual prueba
que $B \sim B \setminus A$.
\end{proof}

\begin{prop}
Sea $X$ un conjunto infinito y sea $A$ un conjunto numerable.
Entonces
\[
X \sim X \cup A.
\]
\end{prop}

\begin{proof}
Si $A \subseteq X$, entonces $X \cup A = X$ y la afirmaciĂłn es trivial.
Supongamos, por lo tanto, que $A$ no estĂˇ contenido en $X$. Definimos
\[
A_0 = A \setminus X,
\]
que es el conjunto de los elementos de $A$ que no pertenecen a $X$.
Como $A$ es numerable, tambiĂ©n $A_0$ es numerable (subconjunto de un
numerable). AdemĂˇs,
\[
X \cup A = X \cup A_0
\]
y la uniĂłn es disjunta, ya que $A_0 \cap X = \varnothing$.

Notemos que $X$ es infinito, luego el conjunto $X \cup A_0$ tambiĂ©n es
infinito. Consideremos ahora el conjunto
\[
B = X \cup A_0.
\]
Entonces $A_0$ es numerable y
\[
B \setminus A_0 = X.
\]
Como $X$ es infinito, tambiĂ©n $B \setminus A_0$ es infinito. Podemos
aplicar la proposiciĂłn anterior con $A = A_0$ y este conjunto $B$:
obtenemos
\[
B \sim B \setminus A_0.
\]
Pero $B = X \cup A_0$ y $B \setminus A_0 = X$, por lo que
\[
X \cup A_0 \sim X.
\]

Como $X \cup A = X \cup A_0$, concluimos que
\[
X \cup A \sim X.
\]
Por simetrĂ­a de la relaciĂłn de equipotencia, tambiĂ©n escribimos
$X \sim X \cup A$, como querĂ­amos.
\end{proof}

\begin{cor}
Un conjunto $X$ es infinito si y sĂłlo si es coordinable con un subconjunto
propio suyo, es decir, si y sĂłlo si existe $Y \subsetneq X$ tal que
$X \sim Y$.
\end{cor}

\begin{proof}
($\Rightarrow$) Supongamos que $X$ es infinito.
Por la proposiciĂłn anterior, existe un subconjunto numerable
$Z \subset X$ tal que $X \sim X \setminus Z$.
Como $Z \ne \varnothing$ (pues es numerable) y $Z \subset X$, se tiene
$X \setminus Z \subsetneq X$.
Por lo tanto, $X$ es coordinable con el subconjunto propio
$X \setminus Z$.

($\Leftarrow$) RecĂ­procamente, supongamos que existe un subconjunto propio
$Y \subsetneq X$ tal que $X \sim Y$.
Procedamos por absurdo: supongamos que $X$ es finito.
Sea $n = \#X$. Entonces $X \sim I_n$.
Como $Y$ es subconjunto propio de $X$, tiene un nĂşmero $m$ de elementos
con $m < n$, de modo que $Y \sim I_m$.

Por transitividad de la relaciĂłn de equipotencia, tendrĂ­amos
\[
I_n \sim X \sim Y \sim I_m,
\]
de donde se seguirĂ­a $I_n \sim I_m$.
Pero por el teorema anterior, $I_n \sim I_m$ implica $n = m$,
lo cual contradice $m < n$.
Esta contradicciĂłn muestra que $X$ no puede ser finito, luego
$X$ es infinito.
\end{proof}

\begin{lema}

\end{lema}

\begin{proof}

\]
Por el teorema fundamental de la aritmĂ©tica, todo nĂşmero natural
tiene una Ăşnica factorizaciĂłn en primos, de modo que distintos pares
$(m,n)$ producen distintos nĂşmeros $2^m 3^n$. Por lo tanto, $f$ es

\end{proof}

\begin{teo}
Sea $(A_n)_{n\in\N}$ una familia de conjuntos numerables.
Entonces la uniĂłn
\[

\]
es a lo sumo numerable. En particular, si $A$ es infinita, entonces
$A$ es numerable.
\end{teo}

\begin{proof}
Como cada $A_n$ es numerable, para todo $n \in \N$ existe una

\]
Escribimos $a_{n,k} = f_n(k)$, de modo que

Para todo $(n,k) \in \N\times\N$ se tiene $a_{n,k} \in A_n \subseteq A$,

Si ademĂˇs $A$ es infinito, por definiciĂłn de â€śa lo sumo numerableâ€ť

\subsection{Cardinalidad de $\R$}

\begin{prop}
$\R \sim (0,1)$.
\end{prop}

\begin{proof}
Veamos primero que $\R \sim (-1,1)$. Consideremos la funciĂłn
\[
f : \R \to (-1,1), \qquad f(x) = \frac{x}{1+|x|}.
\]
Comprobemos que $f$ es biyectiva exhibiendo su inversa
\[
f^{-1} : (-1,1) \to \R, \qquad f^{-1}(y) = \frac{y}{1-|y|}.
\]
Sea $x \in \R$. Como $f(x)$ tiene el mismo signo que $x$, se tiene
$|f(x)| = \dfrac{|x|}{1+|x|}$, de donde
\[
1 - |f(x)| = 1 - \frac{|x|}{1+|x|} = \frac{1}{1+|x|},
\]
y por lo tanto
\[
f^{-1}(f(x)) = \frac{f(x)}{1-|f(x)|}
= \frac{\dfrac{x}{1+|x|}}{\dfrac{1}{1+|x|}} = x.
\]
AnĂˇlogamente, para $y \in (-1,1)$ se tiene $|f^{-1}(y)| = \dfrac{|y|}{1-|y|}$,
de donde $1+|f^{-1}(y)| = \dfrac{1}{1-|y|}$, y por lo tanto
\[
f(f^{-1}(y)) = \frac{f^{-1}(y)}{1+|f^{-1}(y)|}
= \frac{\dfrac{y}{1-|y|}}{\dfrac{1}{1-|y|}} = y.
\]
Como $f^{-1}\circ f = \operatorname{id}_\R$ y $f\circ f^{-1} = \operatorname{id}_{(-1,1)}$,
concluimos que $f$ es biyectiva, y por lo tanto $\R \sim (-1,1)$.

Por otro lado, la funciĂłn
\[
g : (-1,1) \to (0,1), \qquad g(x) = \frac{x+1}{2},
\]
es biyectiva: es inyectiva porque $g(x_1)=g(x_2) \Rightarrow x_1=x_2$, y es
sobreyectiva porque dado $y \in (0,1)$, el elemento $x = 2y-1 \in (-1,1)$
cumple $g(x)=y$. Luego $(-1,1) \sim (0,1)$.

Por transitividad de $\sim$, concluimos que $\R \sim (0,1)$.
\end{proof}

\begin{teo}
El conjunto $\R$ no es numerable.
\end{teo}

\begin{proof}
Por la proposiciĂłn anterior, $\R \sim (0,1)$, asĂ­ que basta probar que
$(0,1)$ no es numerable.

Supongamos, buscando una contradicciĂłn, que $(0,1)$ es numerable, es decir,
que existe una funciĂłn biyectiva $f : \N \to (0,1)$. Escribiendo
$x_n = f(n)$, obtenemos una lista $(x_n)_{n\in\N}$ que enumera todos los
elementos de $(0,1)$, cada uno con un desarrollo decimal
\[

\]

Definimos un nĂşmero $y = 0,y_1y_2y_3\dots \in (0,1)$ eligiendo, para cada
$i \in \N$,
\[

\begin{cases}

\end{cases}
\]
Como todos los dĂ­gitos de $y$ son $2$ o $3$, $y$ tiene un Ăşnico desarrollo

Esta contradicciĂłn muestra que $(0,1)$ no puede ser numerable, y por lo
tanto $\R$ tampoco lo es.
\end{proof}
